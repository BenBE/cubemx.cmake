include(cubemx.cmake/cubemx.cmake)

project(testproject)

add_executable(testproject additional_sources.c)
cubemx_target(
    TARGET testproject
    IOC "${CMAKE_CURRENT_LIST_DIR}/cubemx_project.ioc"
    STARTUP "custom_startup.s"    # optional custom startup
    LDSCRIPT "custom_ldscript.ld" # optional custom ldscript
)
target_compile_options(testproject PRIVATE -Og -Wall -g -gdwarf-2)

# Depending on the project setup, sometimes one of these symbols must be omitted. (Cannot be reliably derived from the .ioc file)
target_compile_definitions(testproject PRIVATE USE_FULL_LL_DRIVER USE_HAL_DRIVER)
